					dfs(v);
				}

	寻找路径（遍历的时候记录路径）
		int depth;
		void dfs(v){
			visit(v); // 标记
			for each u of v{ // 遍取
				depth++;
				if(!visit(u)){ // 避免重复访问
					dfs(v);
				}
				depth--;
			}
			unvisit(v); // 标记为新点，使得其他路径可以访问该点
		}
		void dfs(v, depth){
			visit(v); // 标记
			for each u of v{ // 遍取
				if(!visit(u)){ // 避免重复访问
					dfs(v, depth+1);
				}
			}
			unvisit(v); // 标记为新点
		}

	          怎么知道一个状态可以到哪些状态呢？
